


Write the query to get a list of employees who report to the manager with id 100 (use both first and last name and then use the concatenation with a space in between to make one field as “Employee”)
SELECT FIRST_NAME ||' '|| LAST_NAME AS EMPLOYEE FROM EMPLOYEES WHERE MANAGER_ID=100;
SELECT FIRST_NAME ||' '|| LAST_NAME AS "EMPLOYEE NAME" FROM EMPLOYEES WHERE FIRST_NAME LIKE 'D%'OR LAST_NAME LIKE 'D%';
SELECT FIRST_NAME ||' '|| LAST_NAME AS "EMPLOYEE NAME", MANAGER_ID FROM EMPLOYEES;
SELECT FIRST_NAME ||' '||LAST_NAME  AS NAME ,COMMISSION_PCT AS COMMISSION FROM EMPLOYEES;
Write the query to list employees hired in 1997.
SELECT FIRST_NAME, LAST_NAME, HIRE_DATE FROM EMPLOYEES WHERE HIRE_DATE BETWEEN '1-JANUARY-1997' AND '31-DECEMBER-1997';
SELECT FIRST_NAME,LAST_NAME, SALARY FROM EMPLOYEES WHERE FIRST_NAME='Karen' AND LAST_NAME= 'Partners';
Write the query to find out how much does LISA Ozer make in a year (HINT: MONTH*12 AS Yearly INCOME)?
SELECT FIRST_NAME,LAST_NAME, SALARY*12 "YEARLY INCOME" FROM EMPLOYEES WHERE FIRST_NAME='Lisa' AND LAST_NAME= 'Ozer';
SELECT FIRST_NAME, LAST_NAME, PHONE_NUMBER FROM EMPLOYEES WHERE FIRST_NAME= 'Chen' OR LAST_NAME= 'Chen';
SELECT EMPLOYEE_ID, FIRST_NAME, LAST_NAME,EMAIL FROM EMPLOYEES WHERE EMAIL= 'LPOPP'; 

Front desk clerk needs to know if there is an employee with the name ismaet scialla? [hint: you can use UPPER(), LOWER() or INITCAP() for the query]
SELECT * FROM EMPLOYEES WHERE (FIRST_NAME)='Ismael' AND INITCAP(LAST_NAME)='Sciarra';
Find out who is receiving the Minimum salary?
SELECT FIRST_NAME FROM EMPLOYEES WHERE SALARY= (SELECT MIN(SALARY) FROM EMPLOYEES);
What is the Minimum salary of the office?
SELECT SALARY FROM EMPLOYEES WHERE SALARY= (SELECT MIN(SALARY) FROM EMPLOYEES);
Find out who is receiving the Maximum salary in a year and how much?
SELECT FIRST_NAME,LAST_NAME,SALARY*12 FROM EMPLOYEES WHERE SALARY= (SELECT MAX( SALARY) FROM EMPLOYEES);

SELECT FIRST_NAME, HIRE_DATE FROM EMPLOYEES WHERE HIRE_DATE= (SELECT MIN(HIRE_DATE)FROM EMPLOYEES);

SELECT FIRST_NAME, HIRE_DATE FROM EMPLOYEES WHERE HIRE_DATE= (SELECT MAX(HIRE_DATE)FROM EMPLOYEES);
